El problema
btc_cash = 0.0 btc_position = 0.0 eth_cash = 0.0 # duplicado eth_position = 0.0 # duplicado sol_cash = 0.0 # otra vez... sol_position = 0.0 # otra vez...

N activos = 2N variables + lógica duplicada

La solución
trackers = { "BTC": PositionTracker(), "ETH": PositionTracker(), "SOL": PositionTracker(), } # N activos = N líneas. Cero duplicación.
